def nhap_diem_mon(ten_mon):
    """Nh·∫≠p ƒëi·ªÉm cho m·ªôt m√¥n h·ªçc"""

    """T√≠nh ƒëi·ªÉm trung b√¨nh"""

def hien_thi_bao_cao(ten_hs, diem_cac_mon, diem_tb, xep_loai):

    print("-" * 40)

def phan_tich_diem(diem_cac_mon):

    mon_thap_nhat = [mon for mon, diem in diem_cac_mon.items() if diem == diem_thap_nhat]
    
    print(f"\nüìä PH√ÇN T√çCH CHI TI·∫æT:")

    print(f"   ƒêi·ªÉm th·∫•p nh·∫•t: {diem_thap_nhat} ({', '.join(mon_thap_nhat)})")
    
    # ƒê·∫øm m·ª©c ƒëi·ªÉm
    diem_gioi = sum(1 for diem in diem_cac_mon.values() if diem >= 8.0)
    diem_kha = sum(1 for diem in diem_cac_mon.values() if 6.5 <= diem < 8.0)
    diem_tb = sum(1 for diem in diem_cac_mon.values() if 5.0 <= diem < 6.5)

    print(f"   M√¥n gi·ªèi (‚â•8.0): {diem_gioi}")
    print(f"   M√¥n kh√° (6.5-7.9): {diem_kha}")
    print(f"   M√¥n TB (5.0-6.4): {diem_tb}")
    print(f"   M√¥n y·∫øu (<5.0): {diem_yeu}")

def cong(a, b):
    """Ph√©p c·ªông"""

def may_tinh():
    """M√°y t√≠nh ch√≠nh"""
    print("üßÆ M√ÅY T√çNH ƒê∆†N GI·∫¢N")

    print("C√°c ph√©p t√≠nh: +, -, *, /, **")

            elif phep_tinh == '*':
                ket_qua = nhan(a, b)
            elif phep_tinh == '/':
                ket_qua = chia(a, b)

                ket_qua = luy_thua(a, b)
            else:
                print("‚ùå Ph√©p t√≠nh kh√¥ng h·ªó tr·ª£!")
                continue
            
            print(f"üìä K·∫øt qu·∫£: {bieu_thuc} = {ket_qua}")

            print("‚ùå Vui l√≤ng nh·∫≠p s·ªë h·ª£p l·ªá!")
        except:
            print("‚ùå C√≥ l·ªói x·∫£y ra!")
