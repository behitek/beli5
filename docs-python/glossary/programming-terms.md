```python
# V√≠ d·ª•: Thu·∫≠t to√°n t√¨m s·ªë l·ªõn nh·∫•t trong danh s√°ch
def find_maximum(numbers):
    """Thu·∫≠t to√°n t√¨m s·ªë l·ªõn nh·∫•t."""
    if not numbers:  # N·∫øu danh s√°ch tr·ªëng
```

#### **Class (L·ªõp)**
**Gi·∫£i th√≠ch ƒë∆°n gi·∫£n**: Nh∆∞ b·∫£n thi·∫øt k·∫ø nh√†! üè†

Class l√† "b·∫£n thi·∫øt k·∫ø" ƒë·ªÉ t·∫°o ra c√°c object (ƒë·ªëi t∆∞·ª£ng). Gi·ªëng nh∆∞ b·∫£n v·∫Ω ki·∫øn tr√∫c, t·ª´ m·ªôt class b·∫°n c√≥ th·ªÉ t·∫°o ra nhi·ªÅu object.

```python
class Student:
    """Class Student - b·∫£n thi·∫øt k·∫ø c·ªßa h·ªçc sinh."""
    
    def __init__(self, name, age, student_id):
        """Constructor - h√†m kh·ªüi t·∫°o."""
        self.name = name  # Thu·ªôc t√≠nh (attribute)
        self.age = age
        self.student_id = student_id
        self.grades = []  # Danh s√°ch ƒëi·ªÉm
    
    def add_grade(self, subject, score):
        """Method - ph∆∞∆°ng th·ª©c th√™m ƒëi·ªÉm."""
        grade = {
            'subject': subject,
            'score': score

        self.grades.append(grade)
    
    def get_average(self):
        """Method - t√≠nh ƒëi·ªÉm trung b√¨nh."""
        if not self.grades:
            return 0
        total = sum(grade['score'] for grade in self.grades)
        return total / len(self.grades)
    
    def introduce(self):
        """Method - t·ª± gi·ªõi thi·ªáu."""
        avg = self.get_average()
        return f"Xin ch√†o, t√¥i l√† {self.name}, {self.age} tu·ªïi, MSSV: {self.student_id}, ƒêTB: {avg:.1f}"

# T·∫°o object t·ª´ class
```

#### **Database (C∆° s·ªü d·ªØ li·ªáu)**

```python
    """V√≠ d·ª• v·ªÅ database ƒë∆°n gi·∫£n v·ªõi SQLite."""
    
    def __init__(self, db_name="students.db"):
        self.connection = sqlite3.connect(db_name)

    def create_table(self):
        """T·∫°o b·∫£ng students."""
        self.connection.execute('''
            CREATE TABLE IF NOT EXISTS students (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                name TEXT NOT NULL,
                age INTEGER,
                email TEXT UNIQUE,
                created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP

    def add_student(self, name, age, email):
        """Th√™m h·ªçc sinh v√†o database."""
        try:
            self.connection.execute('''
                INSERT INTO students (name, age, email)
                VALUES (?, ?, ?)
            ''', (name, age, email))
            self.connection.commit()
            print(f"‚úÖ ƒê√£ th√™m {name} v√†o database")
        except sqlite3.IntegrityError:
            print(f"‚ùå Email {email} ƒë√£ t·ªìn t·∫°i!")

        """T√¨m h·ªçc sinh theo t√™n."""
        cursor = self.connection.execute('''
            SELECT * FROM students WHERE name LIKE ?
        ''', (f'%{name}%',))
        
        results = cursor.fetchall()
        return results
    
    def close(self):
        """ƒê√≥ng k·∫øt n·ªëi database."""
        self.connection.close()

# S·ª≠ d·ª•ng database
```

```python
        total = sum(grade['score'] for grade in self.grades)
```
