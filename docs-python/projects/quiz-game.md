class Question:
    """
    Class ƒë·∫°i di·ªán cho m·ªôt c√¢u h·ªèi
    Gi·ªëng nh∆∞ m·ªôt th·∫ª flashcard th√¥ng minh!
    """
    
    def __init__(self, question_text, options, correct_answer, difficulty="medium", 
                 category="general", explanation="", points=10):
        self.question_text = question_text
        self.options = options  # List of 4 options
        self.correct_answer = correct_answer  # Index 0-3
        self.difficulty = difficulty  # easy, medium, hard
        self.category = category
        self.explanation = explanation
        self.points = points
    
    def is_correct(self, answer_index):
        """Ki·ªÉm tra ƒë√°p √°n c√≥ ƒë√∫ng kh√¥ng"""
        return answer_index == self.correct_answer
    
    def get_correct_text(self):
        """L·∫•y text ƒë√°p √°n ƒë√∫ng"""
        return self.options[self.correct_answer]
    
    def to_dict(self):
        """Chuy·ªÉn th√†nh dictionary ƒë·ªÉ l∆∞u file"""
        return {
            'question': self.question_text,
            'options': self.options,
            'correct': self.correct_answer,
            'difficulty': self.difficulty,
            'category': self.category,
            'explanation': self.explanation,
            'points': self.points
        }
    
    @classmethod
    def from_dict(cls, data):
        """T·∫°o Question t·ª´ dictionary"""
        return cls(
            data['question'],
            data['options'],
            data['correct'],
            data.get('difficulty', 'medium'),
            data.get('category', 'general'),
            data.get('explanation', ''),
            data.get('points', 10)
        )

class Player:
    """
    Th√¥ng tin ng∆∞·ªùi ch∆°i
    Gi·ªëng nh∆∞ m·ªôt th·∫ª ng∆∞·ªùi ch∆°i trong game!
    """
    
    def __init__(self, name):
        self.name = name
        self.total_score = 0
        self.games_played = 0
        self.questions_answered = 0
        self.correct_answers = 0
        self.best_score = 0
        self.favorite_category = None
        self.created_date = datetime.now().isoformat()
        self.last_played = None
    
    def update_stats(self, game_score, questions_count, correct_count, category):
        """C·∫≠p nh·∫≠t th·ªëng k√™ ng∆∞·ªùi ch∆°i"""
        self.total_score += game_score
        self.games_played += 1
        self.questions_answered += questions_count
        self.correct_answers += correct_count
        self.last_played = datetime.now().isoformat()
        
        if game_score > self.best_score:
            self.best_score = game_score
        
        # C·∫≠p nh·∫≠t ch·ªß ƒë·ªÅ y√™u th√≠ch (ƒë∆°n gi·∫£n)
        self.favorite_category = category
    
    def get_accuracy(self):
        """T√≠nh ƒë·ªô ch√≠nh x√°c"""
        if self.questions_answered == 0:
            return 0
        return round((self.correct_answers / self.questions_answered) * 100, 1)
    
    def get_average_score(self):
        """T√≠nh ƒëi·ªÉm trung b√¨nh m·ªói game"""
        if self.games_played == 0:
            return 0
        return round(self.total_score / self.games_played, 1)
    
    def to_dict(self):
        """Chuy·ªÉn th√†nh dict ƒë·ªÉ l∆∞u file"""
        return {
            'name': self.name,
            'total_score': self.total_score,
            'games_played': self.games_played,
            'questions_answered': self.questions_answered,
            'correct_answers': self.correct_answers,
            'best_score': self.best_score,
            'favorite_category': self.favorite_category,
            'created_date': self.created_date,
            'last_played': self.last_played
        }
    
    @classmethod
    def from_dict(cls, data):
        """T·∫°o Player t·ª´ dict"""
        player = cls(data['name'])
        player.total_score = data.get('total_score', 0)
        player.games_played = data.get('games_played', 0)
        player.questions_answered = data.get('questions_answered', 0)
        player.correct_answers = data.get('correct_answers', 0)
        player.best_score = data.get('best_score', 0)
        player.favorite_category = data.get('favorite_category')
        player.created_date = data.get('created_date', datetime.now().isoformat())
        player.last_played = data.get('last_played')
        return player

class GameSession:
    """
    Qu·∫£n l√Ω m·ªôt phi√™n ch∆°i game
    Gi·ªëng nh∆∞ m·ªôt tr·ªçng t√†i ƒëi·ªÅu khi·ªÉn game!
    """
    
    def __init__(self, player, questions, time_limit=30):
        self.player = player
        self.questions = questions
        self.time_limit = time_limit
        self.current_question = 0
        self.score = 0
        self.correct_count = 0
        self.start_time = None
        self.hints_used = 0
        self.max_hints = 3
        self.answers_log = []  # L∆∞u log ƒë√°p √°n
        
    def start_game(self):
        """B·∫Øt ƒë·∫ßu game"""
        self.start_time = datetime.now()
        print(f"\n{Fore.GREEN}{Style.BRIGHT}üéÆ B·∫ÆT ƒê·∫¶U GAME!")
        print("=" * 50)
        print(f"üë§ Ng∆∞·ªùi ch∆°i: {self.player.name}")

        input(f"\n{Fore.YELLOW}üì± Nh·∫•n Enter ƒë·ªÉ b·∫Øt ƒë·∫ßu...")
        
    def play_question(self, question_index):
        """Ch∆°i m·ªôt c√¢u h·ªèi"""
        if question_index >= len(self.questions):
            return False
            
        question = self.questions[question_index]
        self.current_question = question_index + 1
        
        print(f"\n{Fore.CYAN}{Style.BRIGHT}")
        print("=" * 60)
        print(f"‚ùì C√ÇU {self.current_question}/{len(self.questions)} - {question.category.upper()}")
        print("=" * 60)
        
        # Hi·ªÉn th·ªã c√¢u h·ªèi
        print(f"{Fore.WHITE}{Style.BRIGHT}{question.question_text}")
        print()
        
        # Hi·ªÉn th·ªã l·ª±a ch·ªçn
        for i, option in enumerate(question.options):
            print(f"{Fore.YELLOW}  {chr(65+i)}. {option}")
        
        print(f"\n{Fore.MAGENTA}üí∞ ƒêi·ªÉm: {question.points} | üí° G·ª£i √Ω c√≤n: {self.max_hints - self.hints_used}")
        
        # ƒê·∫øm ng∆∞·ª£c th·ªùi gian
        answer = self.get_timed_answer()
        
        # Ki·ªÉm tra ƒë√°p √°n
        is_correct = False
        if answer is not None:
            if answer == "hint":
                return self.show_hint(question_index)
            elif 0 <= answer <= 3:
                is_correct = question.is_correct(answer)
        
        # X·ª≠ l√Ω k·∫øt qu·∫£
        self.process_answer(question, answer, is_correct)
        return True
    
    def get_timed_answer(self):
        """L·∫•y ƒë√°p √°n v·ªõi th·ªùi gian gi·ªõi h·∫°n"""
        import select
        import sys
        
        print(f"\n{Fore.GREEN}‚è∞ B·∫°n c√≥ {self.time_limit} gi√¢y ƒë·ªÉ tr·∫£ l·ªùi!")
        print(f"{Fore.CYAN}Nh·∫≠p A, B, C, D ho·∫∑c 'hint' ƒë·ªÉ g·ª£i √Ω: ", end="", flush=True)
        
        start_time = time.time()
        answer = None
        
        # Countdown timer trong thread ri√™ng
        def countdown():
            for remaining in range(self.time_limit, 0, -1):
                time.sleep(1)
                if answer is not None:
                    break
                print(f"\r{Fore.RED}‚è∞ C√≤n {remaining-1} gi√¢y...   ", end="", flush=True)
        
        # B·∫Øt ƒë·∫ßu countdown
        timer_thread = threading.Thread(target=countdown)
        timer_thread.daemon = True
        timer_thread.start()
        
        # L·∫•y input t·ª´ user
        try:
            user_input = input().strip().upper()
            
            if user_input == "HINT":
                answer = "hint"
            elif user_input in ["A", "B", "C", "D"]:
                answer = ord(user_input) - ord("A")  # Chuy·ªÉn A,B,C,D th√†nh 0,1,2,3
            else:
                answer = -1  # Invalid
                
        except:
            answer = None
        
        return answer
    
    def show_hint(self, question_index):
        """Hi·ªÉn th·ªã g·ª£i √Ω"""
        if self.hints_used >= self.max_hints:
            print(f"\n{Fore.RED}‚ùå B·∫°n ƒë√£ h·∫øt l∆∞·ª£t g·ª£i √Ω!")
            return self.play_question(question_index)
        
        self.hints_used += 1
        question = self.questions[question_index]
        
        # Lo·∫°i b·ªè 2 ƒë√°p √°n sai
        correct_index = question.correct_answer
        wrong_indices = [i for i in range(4) if i != correct_index]
        remove_indices = random.sample(wrong_indices, 2)
        
        print(f"\n{Fore.YELLOW}üí° G·ª¢I √ù: Lo·∫°i b·ªè 2 ƒë√°p √°n sai!")
        for i, option in enumerate(question.options):

            else:
                print(f"{Fore.YELLOW}  {chr(65+i)}. {option}")
        
        print(f"\n{Fore.MAGENTA}üí° G·ª£i √Ω c√≤n: {self.max_hints - self.hints_used}")
        
        # Ti·∫øp t·ª•c nh·∫≠n ƒë√°p √°n
        answer = self.get_timed_answer()
        if answer == "hint":
            print(f"\n{Fore.RED}‚ùå B·∫°n ƒë√£ d√πng g·ª£i √Ω r·ªìi!")
            return self.play_question(question_index)
        
        is_correct = False
        if answer is not None and 0 <= answer <= 3:
            is_correct = question.is_correct(answer)
        
        self.process_answer(question, answer, is_correct)
        return True
    
    def process_answer(self, question, answer, is_correct):
        """X·ª≠ l√Ω k·∫øt qu·∫£ ƒë√°p √°n"""
        # Ghi log
        log_entry = {
            'question': question.question_text,
            'user_answer': answer,
            'correct_answer': question.correct_answer,
            'is_correct': is_correct,
            'points': question.points if is_correct else 0
        }
        self.answers_log.append(log_entry)
        
        print(f"\n{'-' * 50}")
        
        if answer is None:
            print(f"{Fore.RED}‚è∞ H·∫æT TH·ªúI GIAN!")
            print(f"{Fore.YELLOW}üí° ƒê√°p √°n ƒë√∫ng: {chr(65 + question.correct_answer)}. {question.get_correct_text()}")
        elif answer == -1:
            print(f"{Fore.RED}‚ùå ƒê√ÅP √ÅN KH√îNG H·ª¢P L·ªÜ!")
            print(f"{Fore.YELLOW}üí° ƒê√°p √°n ƒë√∫ng: {chr(65 + question.correct_answer)}. {question.get_correct_text()}")
        elif is_correct:
            points_earned = question.points
            if self.hints_used > 0:
                points_earned = max(1, points_earned // 2)  # Gi·∫£m ƒëi·ªÉm khi d√πng g·ª£i √Ω
            
            self.score += points_earned
            self.correct_count += 1
            
            print(f"{Fore.GREEN}‚úÖ CH√çNH X√ÅC!")
            print(f"{Fore.GREEN}üéâ +{points_earned} ƒëi·ªÉm")
            print(f"{Fore.CYAN}üí∞ T·ªïng ƒëi·ªÉm: {self.score}")
        else:
            print(f"{Fore.RED}‚ùå SAI R·ªíI!")
            print(f"{Fore.YELLOW}üí° ƒê√°p √°n ƒë√∫ng: {chr(65 + question.correct_answer)}. {question.get_correct_text()}")
        
        # Hi·ªÉn th·ªã gi·∫£i th√≠ch n·∫øu c√≥
        if question.explanation:
            print(f"{Fore.BLUE}üìñ Gi·∫£i th√≠ch: {question.explanation}")
        
        print(f"{'-' * 50}")
        input(f"\n{Fore.YELLOW}üì± Nh·∫•n Enter ƒë·ªÉ ti·∫øp t·ª•c...")
    
    def end_game(self):
